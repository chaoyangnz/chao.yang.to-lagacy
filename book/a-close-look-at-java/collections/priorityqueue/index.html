complete binary tree and array representation

Tree is not necessarily constructed by linked nodes. If we store it as an array, it needs to be a complete binary tree.

A complete binary tree is a binary tree in which every level, except possibly the last, is completely filled, and all nodes are as far left as possible.

So when an array is used to store this tree, a node at n has its two children at 2*n+1 and 2*(n+1).

heap

Heap is a kind of complete binary tree with extra constraints:

  • For a min heap, a node’s value is always smaller than its children. The root node is the none with the minimum value.

  • Likewise for a max heap.

The following always assumes a min heap.

expand

  • Double size if small; else grow by 50%.
  int newCapacity = oldCapacity + ((oldCapacity < 64) ?
                                    (oldCapacity + 2) :
                                    (oldCapacity >> 1));

add node

Node is always initially added to the last, then a shiftUp operation is needed. When shiftUp, the node always compares its value with its parents, if greater than its parent’s value, stop. Otherwise, swap with its parent and then continue this comparison until root.

remove node

If removing a node, this will always happen at root. Once the root is removed, the last node will be used to replace the position. And then a shiftDown operation is needed. Like shiftUp, the node compares its value with its children’s. Swap it with its children with minimum value until the leaf or heap constraint is resolved.

implementation

package me.yangchao;

import java.util.AbstractQueue;
import java.util.Arrays;
import java.util.Iterator;

public class PriorityQueue<E> extends AbstractQueue<E> {

    private static final int DEFAULT_INITIAL_CAPACITY = 11;

    private Object[] queue;
    private int size = 0;

    public PriorityQueue() {
        this.queue = new Object[DEFAULT_INITIAL_CAPACITY];
    }

    @Override
    public int size() {
        return size;
    }

    @Override
    public boolean offer(E e) {
        if(e == null) throw new IllegalArgumentException("priority queue not permit null");
        if(size >= queue.length) {
            resize();
        }
        if(size == 0) {
            queue[0] = e;
        } else {
            queue[size] = e;
            shiftUp(size);
        }

        ++size;
        return true;
    }

    @Override
    public E poll() {
        if(size == 0) return null;
        E root = (E)queue[0];
        if(size == 1) {
            size = 0;
            return root;
        }
        queue[0] = queue[--size]; // replace
        queue[size] = null;
        shiftDown();
        return (E) root;
    }

    private void resize() {
        int capacity = queue.length;
        int newCapacity;
        if(size < 64) {
            newCapacity = capacity * 2; // small capacity -> double
        } else {
            newCapacity = capacity + capacity >>> 1; // large capacity -> increase a half
        }
        queue = Arrays.copyOf(queue, newCapacity);
    }

    private void shiftUp(int i) {
        int j = i;
        while (j > 0) {
            int parent = (j - 1) / 2;
            if(((Comparable)queue[j]).compareTo(queue[parent]) < 0) {//swap
                Object temp = queue[j];
                queue[j] = queue[parent];
                queue[parent] = temp;
                j = parent;
            } else {
                break;
            }
        }
    }

    private void shiftDown() {
        int j = 0;
        while (j < size) {
            Comparable e = (Comparable) queue[j];
            int left = j * 2 + 1;
            int right = j * 2 + 2;
            if(left >= size && right >= size) break;
            if(left < size && right >= size) { // /
                if(e.compareTo(queue[left]) > 0) {
                    Object temp = queue[j];
                    queue[j] = queue[left];
                    queue[left] = temp;
                    j = left;
                } else {
                    break;
                }
            } else { // /\
                int minChild = ((Comparable)queue[left]).compareTo(queue[right]) <= 0 ? left : right;
                if(e.compareTo(queue[minChild]) > 0) {
                    Object temp = queue[j];
                    queue[j] = queue[minChild];
                    queue[minChild] = temp;
                    j = minChild;
                    continue;
                } else {
                    break;
                }
            }
        }
    }

    @Override
    public E peek() {
        if(size == 0) return null;
        return (E) queue[0];
    }

    /////////////////////////////// UNIMPLEMENTED METHODS /////////////////////
    @Override
    public Iterator<E> iterator() {
        return null;
    }
}
